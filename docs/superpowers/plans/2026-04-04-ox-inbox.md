# ox-inbox Crate Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `ox-inbox` crate — a StructFS Store backed by SQLite (metadata) and JSONL files (conversation content) that provides the data layer for an inbox-style agent thread manager.

**Architecture:** InboxStore implements `structfs_core_store::Store` (Reader + Writer). All consumers interact exclusively through StructFS path reads/writes. The store routes metadata operations to SQLite and conversation operations to JSONL files on disk. `Mutex<Connection>` satisfies the `Send + Sync` requirement on Reader/Writer.

**Tech Stack:** Rust (edition 2024), structfs-core-store, structfs-serde-store, rusqlite (bundled), uuid, serde_json, tempfile (dev)

**Spec:** `docs/superpowers/specs/2026-04-04-ox-inbox-design.md`

**Scope:** This plan covers only the ox-inbox crate. Separate plans will cover the TUI redesign and Wasm runtime integration.

```rust
use crate::model::{InboxState, ThreadState};
use rusqlite::Connection;
use std::collections::BTreeMap;
use std::path::PathBuf;
use std::sync::Mutex;
use std::time::{SystemTime, UNIX_EPOCH};
use structfs_core_store::{Error, Path, Record, Value};

fn now_epoch() -> i64 {
    SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap()
        .as_secs() as i64
}

fn extract_string(map: &BTreeMap<String, Value>, key: &str) -> Option<String> {
    match map.get(key) {
        Some(Value::String(s)) => Some(s.clone()),
        _ => None,
    }
}

fn extract_string_array(map: &BTreeMap<String, Value>, key: &str) -> Vec<String> {
    match map.get(key) {
        Some(Value::Array(arr)) => arr
            .iter()
            .filter_map(|v| match v {
                Value::String(s) => Some(s.clone()),
                _ => None,
            })
            .collect(),
        _ => vec![],
    }
}

pub fn write_dispatch(
    db: &Mutex<Connection>,
    threads_dir: &PathBuf,
    to: &Path,
    data: Record,
) -> Result<Path, Error> {
    let components: Vec<&String> = to.iter().collect();

    match components.as_slice() {
        [first] if first.as_str() == "threads" => create_thread(db, threads_dir, data),
        [first, id] if first.as_str() == "threads" => update_thread(db, id, data),
        [first, id, sub] if first.as_str() == "threads" && sub.as_str() == "labels" => {
            set_labels(db, id, data)
        }
        [first, id, sub] if first.as_str() == "threads" && sub.as_str() == "messages" => {
            append_message(threads_dir, id, data)
        }
        [first, id, sub] if first.as_str() == "threads" && sub.as_str() == "tasks" => {
            create_task(db, id, data)
        }
        [first, id, sub, task_id]
            if first.as_str() == "threads" && sub.as_str() == "tasks" =>
        {
            update_task(db, id, task_id, data)
        }
        _ => Err(Error::store(
            "InboxStore",
            "write",
            format!("unrecognized path: {}", to),
        )),
    }
}

fn create_thread(
    db: &Mutex<Connection>,
    threads_dir: &PathBuf,
    data: Record,
) -> Result<Path, Error> {
    let value = data
        .as_value()
        .ok_or_else(|| Error::store("InboxStore", "write", "expected parsed record"))?;
    let Value::Map(ref map) = value else {
        return Err(Error::store("InboxStore", "write", "expected map value"));
    };

    let title = extract_string(map, "title")
        .ok_or_else(|| Error::store("InboxStore", "write", "missing title"))?;
    let parent_id = extract_string(map, "parent_id");
    let labels = extract_string_array(map, "labels");

    let id = uuid::Uuid::new_v4().to_string();
    let now = now_epoch();

    // Create JSONL directory for this thread
    let thread_dir = threads_dir.join(&id);
    std::fs::create_dir_all(&thread_dir).map_err(Error::from)?;

    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "write", format!("lock poisoned: {}", e))
    })?;

    conn.execute(
        "INSERT INTO threads (id, title, parent_id, inbox_state, thread_state, created_at, updated_at)
         VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7)",
        rusqlite::params![
            id,
            title,
            parent_id,
            InboxState::Inbox.as_str(),
            ThreadState::Running.as_str(),
            now,
            now,
        ],
    )
    .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;

    for label in &labels {
        conn.execute(
            "INSERT INTO labels (thread_id, label) VALUES (?1, ?2)",
            rusqlite::params![id, label],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    Path::parse(&format!("threads/{}", id)).map_err(Error::from)
}

fn update_thread(db: &Mutex<Connection>, id: &str, data: Record) -> Result<Path, Error> {
    let value = data
        .as_value()
        .ok_or_else(|| Error::store("InboxStore", "write", "expected parsed record"))?;
    let Value::Map(ref map) = value else {
        return Err(Error::store("InboxStore", "write", "expected map value"));
    };

    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "write", format!("lock poisoned: {}", e))
    })?;
    let now = now_epoch();

    if let Some(title) = extract_string(map, "title") {
        conn.execute(
            "UPDATE threads SET title = ?1, updated_at = ?2 WHERE id = ?3",
            rusqlite::params![title, now, id],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    if let Some(inbox_state) = extract_string(map, "inbox_state") {
        if InboxState::parse(&inbox_state).is_none() {
            return Err(Error::store(
                "InboxStore",
                "write",
                format!("invalid inbox_state: {}", inbox_state),
            ));
        }
        conn.execute(
            "UPDATE threads SET inbox_state = ?1, updated_at = ?2 WHERE id = ?3",
            rusqlite::params![inbox_state, now, id],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    if let Some(thread_state) = extract_string(map, "thread_state") {
        if ThreadState::parse(&thread_state).is_none() {
            return Err(Error::store(
                "InboxStore",
                "write",
                format!("invalid thread_state: {}", thread_state),
            ));
        }
        conn.execute(
            "UPDATE threads SET thread_state = ?1, updated_at = ?2 WHERE id = ?3",
            rusqlite::params![thread_state, now, id],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    if let Some(reason) = extract_string(map, "block_reason") {
        conn.execute(
            "UPDATE threads SET block_reason = ?1, updated_at = ?2 WHERE id = ?3",
            rusqlite::params![reason, now, id],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    if let Some(Value::Integer(count)) = map.get("token_count") {
        conn.execute(
            "UPDATE threads SET token_count = ?1, updated_at = ?2 WHERE id = ?3",
            rusqlite::params![count, now, id],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    Path::parse(&format!("threads/{}", id)).map_err(Error::from)
}

fn set_labels(db: &Mutex<Connection>, id: &str, data: Record) -> Result<Path, Error> {
    let value = data
        .as_value()
        .ok_or_else(|| Error::store("InboxStore", "write", "expected parsed record"))?;
    let Value::Array(ref arr) = value else {
        return Err(Error::store(

    };

        .iter()

            Value::String(s) => Some(s.clone()),
            _ => None,
        })
        .collect();

        Error::store("InboxStore", "write", format!("lock poisoned: {}", e))
    })?;

    conn.execute(
        "DELETE FROM labels WHERE thread_id = ?1",
        rusqlite::params![id],
    )
    .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;

    for label in &labels {
        conn.execute(
            "INSERT INTO labels (thread_id, label) VALUES (?1, ?2)",
            rusqlite::params![id, label],
        )
        .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
    }

    let now = now_epoch();
    conn.execute(
        "UPDATE threads SET updated_at = ?1 WHERE id = ?2",
        rusqlite::params![now, id],
    )
    .map_err(|e| Error::store("InboxStore", "write", e.to_string()))?;
```

```rust
use crate::model::{InboxState, TaskInfo, ThreadMetadata, ThreadState};
use rusqlite::Connection;
use std::path::PathBuf;
use std::sync::Mutex;
use structfs_core_store::{Error, Path, Record, Value};

pub fn read_dispatch(
    db: &Mutex<Connection>,
    threads_dir: &PathBuf,
    from: &Path,
) -> Result<Option<Record>, Error> {
    let components: Vec<&String> = from.iter().collect();

    match components.as_slice() {
        [] => Ok(None),
        [first] if first.as_str() == "threads" => list_threads(db, "inbox"),
        [first] if first.as_str() == "done" => list_threads(db, "done"),
        [first] if first.as_str() == "labels" => list_labels(db),
        [first, id] if first.as_str() == "threads" => get_thread(db, id),
        [first, name] if first.as_str() == "labels" => threads_by_label(db, name),
        [first, state] if first.as_str() == "by_state" => threads_by_state(db, state),
        [first, query] if first.as_str() == "search" => search_threads(db, query),
        [first, id, sub] if first.as_str() == "threads" && sub.as_str() == "messages" => {
            read_messages(threads_dir, id)
        }
        [first, id, sub] if first.as_str() == "threads" && sub.as_str() == "children" => {
            list_children(db, id)
        }
        [first, id, sub] if first.as_str() == "threads" && sub.as_str() == "tasks" => {
            list_tasks(db, id)
        }
        _ => Ok(None),
    }
}

fn query_threads(
    conn: &Connection,
    where_clause: &str,
    params: &[&dyn rusqlite::types::ToSql],
) -> Result<Vec<ThreadMetadata>, Error> {
    let sql = format!(
        "SELECT id, title, parent_id, inbox_state, thread_state, block_reason,
                created_at, updated_at, token_count
         FROM threads
         WHERE {}
         ORDER BY updated_at DESC",
        where_clause
    );
    let mut stmt = conn
        .prepare(&sql)
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;
    let rows = stmt
        .query_map(params, |row| {
            Ok(ThreadMetadata {
                id: row.get(0)?,
                title: row.get(1)?,
                parent_id: row.get(2)?,
                inbox_state: InboxState::parse(
                    &row.get::<_, String>(3)?,
                )
                .unwrap_or(InboxState::Inbox),
                thread_state: ThreadState::parse(
                    &row.get::<_, String>(4)?,
                )
                .unwrap_or(ThreadState::Running),
                block_reason: row.get(5)?,
                created_at: row.get(6)?,
                updated_at: row.get(7)?,
                token_count: row.get(8)?,
                labels: vec![], // filled below
            })
        })
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;

    let mut threads: Vec<ThreadMetadata> = rows
        .collect::<Result<_, _>>()
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;

    // Batch-load labels for all threads
    for thread in &mut threads {
        thread.labels = query_labels(conn, &thread.id)?;
    }

    Ok(threads)
}

fn query_labels(conn: &Connection, thread_id: &str) -> Result<Vec<String>, Error> {
    let mut stmt = conn
        .prepare("SELECT label FROM labels WHERE thread_id = ?1 ORDER BY label")
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;
    let labels = stmt
        .query_map([thread_id], |row| row.get(0))
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?
        .collect::<Result<Vec<String>, _>>()
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;
    Ok(labels)
}

fn threads_to_value(threads: Vec<ThreadMetadata>) -> Option<Record> {
    let arr: Vec<Value> = threads.iter().map(|t| t.to_value()).collect();
    Some(Record::parsed(Value::Array(arr)))
}

fn list_threads(
    db: &Mutex<Connection>,
    inbox_state: &str,
) -> Result<Option<Record>, Error> {
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let threads = query_threads(&conn, "inbox_state = ?1", &[&inbox_state])?;
    Ok(threads_to_value(threads))
}

fn get_thread(db: &Mutex<Connection>, id: &str) -> Result<Option<Record>, Error> {
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let mut threads = query_threads(&conn, "id = ?1", &[&id])?;
    match threads.len() {
        0 => Ok(None),
        _ => Ok(Some(Record::parsed(threads.remove(0).to_value()))),
    }
}

fn list_children(
    db: &Mutex<Connection>,
    parent_id: &str,

    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let threads = query_threads(&conn, "parent_id = ?1", &[&parent_id])?;
    Ok(threads_to_value(threads))
}

fn threads_by_label(
    db: &Mutex<Connection>,
    label: &str,
) -> Result<Option<Record>, Error> {
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let threads = query_threads(
        &conn,
        "id IN (SELECT thread_id FROM labels WHERE label = ?1) AND inbox_state = 'inbox'",
        &[&label],
    )?;
    Ok(threads_to_value(threads))
}

fn threads_by_state(
    db: &Mutex<Connection>,
    state: &str,
) -> Result<Option<Record>, Error> {
    if ThreadState::parse(state).is_none() {
        return Ok(Some(Record::parsed(Value::Array(vec![]))));
    }
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let threads = query_threads(
        &conn,
        "thread_state = ?1 AND inbox_state = 'inbox'",
        &[&state],
    )?;
    Ok(threads_to_value(threads))
}

fn search_threads(
    db: &Mutex<Connection>,
    query: &str,
) -> Result<Option<Record>, Error> {
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let pattern = format!("%{}%", query);
    let threads = query_threads(
        &conn,
        "inbox_state = 'inbox' AND (title LIKE ?1 OR id IN (SELECT thread_id FROM labels WHERE label LIKE ?1))",
        &[&pattern as &dyn rusqlite::types::ToSql],
    )?;
    Ok(threads_to_value(threads))
}

fn list_labels(db: &Mutex<Connection>) -> Result<Option<Record>, Error> {
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let mut stmt = conn
        .prepare("SELECT DISTINCT label FROM labels ORDER BY label")
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;
    let labels: Vec<Value> = stmt
        .query_map([], |row| {
            let l: String = row.get(0)?;
            Ok(Value::String(l))
        })
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?
        .collect::<Result<_, _>>()
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;
    Ok(Some(Record::parsed(Value::Array(labels))))
}

fn list_tasks(
    db: &Mutex<Connection>,
    thread_id: &str,
) -> Result<Option<Record>, Error> {
    let conn = db.lock().map_err(|e| {
        Error::store("InboxStore", "read", format!("lock poisoned: {}", e))
    })?;
    let mut stmt = conn
        .prepare(
            "SELECT id, thread_id, title, status, created_at, updated_at
             FROM tasks WHERE thread_id = ?1 ORDER BY created_at",
        )
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?;
    let tasks: Vec<Value> = stmt
        .query_map([thread_id], |row| {
            Ok(TaskInfo {
                id: row.get(0)?,
                thread_id: row.get(1)?,
                title: row.get(2)?,
                status: row.get(3)?,
                created_at: row.get(4)?,
                updated_at: row.get(5)?,
            })
        })
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?
        .collect::<Result<Vec<TaskInfo>, _>>()
        .map_err(|e| Error::store("InboxStore", "read", e.to_string()))?
        .into_iter()
        .map(|t| t.to_value())
        .collect();
    Ok(Some(Record::parsed(Value::Array(tasks))))
}

fn read_messages(
    threads_dir: &PathBuf,
    id: &str,
) -> Result<Option<Record>, Error> {
    let jsonl_path = threads_dir.join(id).join(format!("{}.jsonl", id));
    let lines = crate::jsonl::read_all(&jsonl_path)?;
    let messages: Vec<Value> = lines
        .into_iter()
        .filter_map(|line| {
            serde_json::from_str::<serde_json::Value>(&line)
                .ok()
                .map(structfs_serde_store::json_to_value)
        })
        .collect();
    Ok(Some(Record::parsed(Value::Array(messages))))
}
```
